CREATE TABLE ROOT.ANSWERS (
ID INT NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
IDQUESTION INT,
IDUSER INT, 
TEXT VARCHAR(5000),
DATEADD DATE,
PRIMARY KEY (ID)
);


CREATE TABLE ROOT.CATEGORY (
ID INT NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
NAME VARCHAR(250),
PRIMARY KEY (ID)
);

CREATE TABLE ROOT.QUESTIONS (
ID INT NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
NAME VARCHAR(250),
IDCATEGORY INT,
IDUSER INT, 
FULLLTEXT VARCHAR(5000),
DATEADD DATE,
PRIMARY KEY (ID)
);

CREATE TABLE ROOT.TYPEUSERS (
ID INT NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
NAME VARCHAR(250),
PRIMARY KEY (ID)
);




CREATE TABLE ROOT.USERS (
ID INT NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
USERNAME VARCHAR(250),
IDTYPEUSER INT,
EMAIL VARCHAR(250),
PICTURE VARCHAR(500),
PASSWORD VARCHAR(500),
PRIMARY KEY (ID)
);
